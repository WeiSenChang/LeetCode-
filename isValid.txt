//有效的括号
bool isValid(string s) {
	stack<char> st;//栈，存放左括号
	map<char, char>m = { {'{','}'},{'[',']'},{'(',')'} };//哈希表，key为左括号，value为对应的右括号
	for (int i = 0; i < s.length(); i++) {
		//当栈不为空，进行判断
		if (!st.empty()) {
			if (m[st.top() == s[i]]) {//如果遍历到左括号对应的右括号
				st.pop();//将栈顶移出
				continue;//跳过压栈，进入下一个循环
			}
		}
		//将左括号压入栈中
		st.push(s[i]);
	}
	return st.empty();
}
